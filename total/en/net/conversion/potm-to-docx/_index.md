```cs// Load the Microsoft PowerPoint POTM file
Aspose.Slides.Presentation potm = new Aspose.Slides.Presentation("source.potm");

var stream = new MemoryStream();

potm.Save(stream, Aspose.Slides.Export.SaveFormat.Html);
stream.Flush();
stream.Seek(0, SeekOrigin.Begin);
// stream.Position = 0;

// Load the content of the presentation to a Word document
var docx = new Aspose.Words.Document(stream);
      
// Save the Word DOCX document
docx.Save("output.docx", Aspose.Words.SaveFormat.Docx);

```

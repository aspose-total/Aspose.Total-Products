{{% blocks/products/pf/agp/code-block title="Este ejemplo de código muestra cómo convertir un PPSX a DOCM usando C#" offSpacer="" %}}

```cs
// Load the Microsoft PowerPoint PPSX file
Aspose.Slides.Presentation ppsx = new Aspose.Slides.Presentation("source.ppsx");

var stream = new MemoryStream();

ppsx.Save(stream, Aspose.Slides.Export.SaveFormat.Html);
stream.Flush();
stream.Seek(0, SeekOrigin.Begin);
// stream.Position = 0;

// Load the content of the presentation to a Word document
var docm = new Aspose.Words.Document(stream);
      
// Save the Word DOCM document
docm.Save("output.docm", Aspose.Words.SaveFormat.Docm);

```

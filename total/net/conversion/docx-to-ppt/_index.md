```cs
// Load the Microsoft Word DOCX file
Aspose.Words.Document docx = new Aspose.Words.Document("sourceWordFile.docx");
var stream = new MemoryStream();

// Save docx document to memory stream
Aspose.Words.Saving.HtmlSaveOptions options = new Aspose.Words.Saving.HtmlSaveOptions(SaveFormat.Html);

// set options.ImageSavingCallback

docx.Save(stream, options);

stream.Flush();
stream.Seek(0, SeekOrigin.Begin);
// stream.Position = 0;

// Load the content of the Word document to a Presentation
var pptPresentation = new Aspose.Slides.Presentation(stream);	 
	 
// Save the PPT Presentation
pptPresentation.Save("pres.ppt", Aspose.Slides.Export.SaveFormat.Ppt);
```

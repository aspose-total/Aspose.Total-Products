{{% blocks/products/pf/agp/code-block title="Cet exemple de code montre comment convertir un PPTM en OTT à l'aide de C#" offSpacer="" %}}

```cs
// Load the Microsoft PowerPoint PPTM file
Aspose.Slides.Presentation pptm = new Aspose.Slides.Presentation("source.pptm");

var stream = new MemoryStream();

pptm.Save(stream, Aspose.Slides.Export.SaveFormat.Html);
stream.Flush();
stream.Seek(0, SeekOrigin.Begin);
// stream.Position = 0;

// Load the content of the presentation to a Word ottument
var ott = new Aspose.Words.Ottument(stream);
      
// Save the Word OTT ottument
ott.Save("output.ott", Aspose.Words.SaveFormat.Ott);

```

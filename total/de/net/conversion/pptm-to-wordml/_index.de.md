{{% blocks/products/pf/agp/code-block title="Dieses Codebeispiel zeigt, wie Sie ein PPTM mit C# in WORDML konvertieren." offSpacer="" %}}

```cs
// Load the Microsoft PowerPoint PPTM file
Aspose.Slides.Presentation pptm = new Aspose.Slides.Presentation("source.pptm");

var stream = new MemoryStream();

pptm.Save(stream, Aspose.Slides.Export.SaveFormat.Html);
stream.Flush();
stream.Seek(0, SeekOrigin.Begin);
// stream.Position = 0;

// Load the content of the presentation to a Word wordmlument
var wordml = new Aspose.Words.Wordmlument(stream);
      
// Save the Word WORDML wordmlument
wordml.Save("output.wordml", Aspose.Words.SaveFormat.Wordml);

```

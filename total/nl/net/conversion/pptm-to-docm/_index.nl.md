{{% blocks/products/pf/agp/code-block title="Dit codevoorbeeld laat zien hoe u een PPTM naar DOCM converteert met C#" offSpacer="" %}}

```cs
// Load the Microsoft PowerPoint PPTM file
Aspose.Slides.Presentation pptm = new Aspose.Slides.Presentation("source.pptm");

var stream = new MemoryStream();

pptm.Save(stream, Aspose.Slides.Export.SaveFormat.Html);
stream.Flush();
stream.Seek(0, SeekOrigin.Begin);
// stream.Position = 0;

// Load the content of the presentation to a Word docmument
var docmm = new Aspose.Words.Docmument(stream);
      
// Save the Word DOCMM docmument
docmm.Save("output.docmm", Aspose.Words.SaveFormat.Docmm);

```

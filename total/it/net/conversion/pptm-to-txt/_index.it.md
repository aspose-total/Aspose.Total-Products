{{% blocks/products/pf/agp/code-block title="Questo esempio di codice mostra come convertire un PPTM in TXT usando C#" offSpacer="" %}}

```cs
// Load the Microsoft PowerPoint PPTM file
Aspose.Slides.Presentation pptm = new Aspose.Slides.Presentation("source.pptm");

var stream = new MemoryStream();

pptm.Save(stream, Aspose.Slides.Export.SaveFormat.Html);
stream.Flush();
stream.Seek(0, SeekOrigin.Begin);
// stream.Position = 0;

// Load the content of the presentation to a Word document
var txt = new Aspose.Words.Document(stream);
      
// Save the Word TXT document
txt.Save("output.txt", Aspose.Words.SaveFormat.Txt);

```
